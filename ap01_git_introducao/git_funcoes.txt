# Criando repositório e configurando git

git init 
    (cria a pasta .git)
git branch -m main
    (-m: renomiando a branch atual para "main")
git config --global init.defaultBranch main
    (todas as branchs criadas no SO atual serao criadas [git init] ja com o nome "main")
git config --global core.editor "code --wait"
    (caso um comando necessite de um editor de texto, uma aba sera aberta no VSCode [code] ao invés de na linha de comando. E faz o Code esperar algo ser digitado antes de fechar a aba)
git config --list
    (lista as configurações, sai com Q, avança com ENTER)
git config --list --show-origin
git config --unset core.editor
git config (--gloabal) --edit
git status (--short)

# Estados dos arquivos

// todo arquivo passa pelo UNTRACKED

: UNTRACKED
    -> Fora do controle de versão
    git rm (transforma o arquivo em UNTRACKED)
git add (o que)
: TRACKED
    -> No controle de versão
    : UNMODIFIED
        -> Não modificado
    : MODIFIED
        -> Modificado e diferente do permante
    git add -> vai para STAGED
    : STAGED
        -> Participara do próximo commit

# Tornando arquivo permanente

git add .(adiciona tudo)

# TAG

git show (nome da tag)

criando tags: 
 - leve (rotulo interno [não é release -> temporaria])
 - anotada

git tag -a(anotated) v1.0.0 -m (mensagem)
    (versionamento semântico)
    MAJOR.MINOR.PATCH

    v1.0.0
    bug arrumado
    v1.0.1
    adicionou nova funcionalidade
    v1.1.0 (patch da minor 1 ainda é zero)
    subtração aprimorada em desempenho
    v1.1.1
    modifica o modo de utilizar uma funcionalidade previa (breaking change)
        -> muda o jeito de se utilizar
        precisa passar uma manual de transição para o usuario
    v2.0.0
    "soma" passou  a ser chamada "sum" (breaking change)
    v3.0.0
